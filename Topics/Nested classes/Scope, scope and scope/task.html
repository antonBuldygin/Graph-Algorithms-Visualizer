What will be the output of this code?

public class Outer {

    private int x = 5;

    void someMethod() {

        final int x = 10;

        public class LocalInner {

            private void print() {
                System.out.println("x = " + x);
            }
        }

        LocalInner inner = new LocalInner();
        inner.print();
    }

    public static void main(String[] args) {
        Outer outer = new Outer();
        outer.someMethod();
    }
}
